```typescript
        if (cookie.parselet_index >= parselets.length) {
          // No parselet matched, parse error
          task.out(makeErrorThing('No parselet matched'));
          return;
        }

        const parselet = parselets[cookie.parselet_index];
        cookie.parselet_index++;

        if (isNativeFunction(parselet)) {
          // Call native parselet directly
          const result = callNativeParselet(
            parselet,
            cookie.tokens,
            cookie.position,
            cookie.env
          );
          cookie.left_expr = result.expr;
          cookie.position = result.position;
          cookie.phase = 'got_prefix_result';
          task.updateCookie(0, 0, cookie);
          return;
        } else {
          // Call Backolon parselet
          const parseletCode = parselet; // It's a Backolon function
          const parseletArgs = [
            makeListThing(cookie.tokens),
            makeNumberThing(cookie.position),
            cookie.env
          ];

          cookie.phase = 'got_prefix_result';
          task.updateCookie(0, 0, cookie);
          task.enter(
            parseletCode,
            state.loc,
            cookie.env,
            parseletArgs,
            'parselet'
          );
          return;
        }
      }

      case 'got_prefix_result': {
        // The parselet returned (or the Backolon call returned)
        const result = task.result;

        // Parse the result: expect { expr, position } or call failed
        if (isErrorThing(result)) {
          // Parselet failed, try next
          cookie.phase = 'try_parselet';
          task.updateCookie(0, 0, cookie);
          return;
        }

        cookie.left_expr = getField(result, 'expr');
        cookie.position = coerceToNumber(getField(result, 'position'));
        cookie.phase = 'try_infix';
        task.updateCookie(0, 0, cookie);
        return;
      }

      case 'try_infix': {
        // Check if there's an infix operator at the current position
        if (cookie.position >= cookie.tokens.length) {
          // End of input
          cookie.phase = 'done';
          task.updateCookie(0, 0, cookie);
          return;
        }

        const currentToken = cookie.tokens[cookie.position];
        const infixParselets = getInfixParselets(
          cookie.precedence_table,
          currentToken,
          cookie.min_prec
        );

        if (infixParselets.length === 0) {
          // No infix operator, we're done
          cookie.phase = 'done';
          task.updateCookie(0, 0, cookie);
          return;
        }

        // Try first infix parselet
        cookie.parselet_index = 0;
        const parselet = infixParselets[0];

        if (isNativeFunction(parselet)) {
          const result = callNativeParselet(
            parselet,
            cookie.tokens,
            cookie.position,
            cookie.env,
            cookie.left_expr
          );
          cookie.left_expr = result.expr;
          cookie.position = result.position;
          task.updateCookie(0, 0, cookie);
          return; // Loop back to try_infix
        } else {
          const parseletArgs = [
            makeListThing(cookie.tokens),
            makeNumberThing(cookie.position),
            cookie.env,
            cookie.left_expr
          ];

          cookie.phase = 'got_infix_result';
          task.updateCookie(0, 0, cookie);
          task.enter(
            parselet,
            state.loc,
            cookie.env,
            parseletArgs,
            'infix_parselet'
          );
          return;
        }
      }

      case 'got_infix_result': {
        const result = task.result;

        if (isErrorThing(result)) {
          // Infix parselet failed, no more infixes
          cookie.phase = 'done';
          task.updateCookie(0, 0, cookie);
          return;
        }

        cookie.left_expr = getField(result, 'expr');
        cookie.position = coerceToNumber(getField(result, 'position'));
        cookie.phase = 'try_infix';
        task.updateCookie(0, 0, cookie);
        return;
      }

      case 'done': {
        // Return the final result
        const result = makeMapThing({
          expr: cookie.left_expr,
          position: makeNumberThing(cookie.position)
        });
        task.out(result);
        return;
      }
    }
  }
);
```

**This implementation:**

1. **Uses the cookie system** to maintain parser state across multiple invocations
2. **Separates prefix and infix parsing** phases as separate cookie states
3. **Handles both native and Backolon parselets** — native ones are called directly, Backolon ones go through `task.enter()`
4. **Is fully re-entrant** — the parser state is explicit data, not implicit in the call stack
5. **Never corrupts when continuations happen** — when a parselet calls `call/cc`, the continuation captures the `ParserCookie` as immutable data

The key: parselets should return `{expr: <result>, position: <new_pos>}` or an error thing on failure. When you call a parselet, pass it the tokens, current position, and environment, and it updates the position as it consumes tokens.
